| Aula 04 - JavaScript (JS)


<script>
        const itens = document.getElementById('itens');

        function clickMenu() {
            if (itens.style.display == 'block') {
                itens.style.display = 'none';
            } else {
                itens.style.display = 'block';
            }
        }
        
        // Fecha o menu automaticamente ao clicar em um link
        const links = document.querySelectorAll('#itens a');
        links.forEach(link => {
            link.addEventListener('click', () => {
                // Apenas fecha se estiver em tela menor que 768px (Mobile)
                if (window.innerWidth < 768) { 
                    itens.style.display = 'none';
                }
            });
        });

        function mudouTamanho() {
            if (window.innerWidth >= 768) {
                // Desktop: sempre mostra o menu
                itens.style.display = 'block';
            } else {
// Mobile: esconde o menu (deixa apenas o botão)
                itens.style.display = 'none';
            }
        }


// Garante que a função seja chamada na primeira carga para definir o estado inicial
mudouTamanho();

    </script>
</body>
